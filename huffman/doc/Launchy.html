def self.open(link)
    if OS.windows?
        system "start #{link}"
    elsif OS.mac?
        system "open #{link}"
    elsif OS.linux? or OS.bsd?
        system "xdg-open #{link}"
    else
        STDERR.puts "Don't recognize your operating system."
        STDERR.puts "Guessing Windows, trying to open browser."
        system "start #{link}"
    end
end
