    Categorize the host OS as Windows, Mac, Linux, BSD, or unknown.

    # File hack_os.rb, line 12
def self.os_category
    if OS.windows? then
        "Windows"
    elsif OS.mac?
        "Mac OS"
    elsif OS.linux?
        "Linux"
    elsif OS.bsd?
        "BSD Unix"
    else
        "UNKNOWN"
    end
end
